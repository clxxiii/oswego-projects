6    package npw;
7    
8    import painter.SPainter;
9    import shapes.SRectangle;
10   
11   import javax.swing.*;
12   import java.awt.*;
13   
14   public class GreenT {
15       // Green T code
16       private void paintTheImage() {
17           // Make Painter
18           SPainter klee = new SPainter("Green T", 600, 600);
19           // Make Rectangle
20           SRectangle rect = new SRectangle(400,100);
21           // Move the painter down to make room for the T
22           klee.setHeading(180); klee.mfd(50);
23           // Make painter paint in green
24           klee.setColor(Color.GREEN);
25           //Paint rectangle for the first time
26           klee.paint(rect);
27   
28           // Change the dimensions of the rectangle
29           rect.setHeight(100); rect.setWidth(400);
30           //Move painter up (the painter is already facing down, so I just move it backwards)
31           klee.mbk(250);
32           //Paint the rectangle again
33           klee.paint(rect);
34       }
35   
36       // call paintTheImage() upon construction
37       public GreenT() {
38           paintTheImage();
39       }
40   
41       // Construct a blue dot when calling main()
42       public static void main(String[] args) {
43           SwingUtilities.invokeLater(new Runnable() {
44               public void run() {
45                   new GreenT();
46               }
47           });
48       }
49   }
