6    public class Primes {
7        public static void main(String[] args) {

9    
10           primes[0] = 2;
11           primes[1] = 3;
12           primes[2] = 5;
13           primes[3] = 7;
14   
15           System.out.println("length of primes array = " + primes.length);
16           System.out.println("first prime = " + primes[0]);
17           System.out.println("last prime = " + primes[3]);
18           System.out.println("last prime = " + primes[primes.length - 1]);
19   
20           System.out.println("\nThe inital array ...");
21           int i = 0;
22           while (i < primes.length) {
23               System.out.println(primes[i]);
24               i = i + 1;
25           }
26   
27           int temp = primes[0];
28           primes[0] = primes[primes.length - 1];
29           primes[primes.length - 1] = temp;
30   
31           System.out.println("\nThe final array ...");
32           for (int x = 0; x < primes.length; x = x + 1) {
33               System.out.println(primes[x]);
34           }
35       }
36   }
