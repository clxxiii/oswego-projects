6    package npw;

8    import painter.SPainter;
9    import shapes.SCircle;
10   
11   import javax.swing.*;
12   import java.awt.*;
13   
14   public class Target {
15       // Target Code
16       private void paintTheImage() {
17           // Make the painter
18           SPainter klee = new SPainter("Target", 600, 600);
19           // Make the three dots
20           SCircle outerDot = new SCircle(150);
21           SCircle middleDot = new SCircle(100);
22           SCircle innerDot = new SCircle(50);
23           // Paint dot 1
24           klee.setColor(Color.RED);
25           klee.paint(outerDot);
26           // Paint dot 2
27           klee.setColor(Color.WHITE);
28           klee.paint(middleDot);
29           // Paint dot 3
30           klee.setColor(Color.RED);
31           klee.paint(innerDot);
32       }
33   
34       // call paintTheImage() upon construction
35       public Target() {
36           paintTheImage();
37       }
38   
39       // Construct a blue dot when calling main()
40       public static void main(String[] args) {
41           SwingUtilities.invokeLater(new Runnable() {
42               public void run() {
43                   new Target();
44               }
45           });
46       }
