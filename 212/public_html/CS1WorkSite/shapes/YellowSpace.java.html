2     * A program for calculating the yellow space on the diagram in Challenge 3 
3     */
4    package shapes;
5    
6    public class YellowSpace {
7        public static void main(String[] args) {
8            int tableclothSideLength = 800;
9            int diamondDistanceFromTablecloth = 80;
10           int squareDistanceFromDiamond = 40;
11   
12           // Create the yellow diamond
13           double diamondCircleDiameter = tableclothSideLength - (diamondDistanceFromTablecloth * 2);
14           SCircle diamondCircle = new SCircle(diamondCircleDiameter / 2);
15           SSquare diamond = diamondCircle.inscribingSquare();
16   
17           // Create the square
18           double squareCircleDiameter = diamond.side() - (squareDistanceFromDiamond * 2);
19           SCircle squareCircle = new SCircle(squareCircleDiameter / 2);
20           SSquare greySquare = squareCircle.inscribingSquare();
21   
22           // Calculate and print the yellow space
23           double yellowSpace = diamond.area() - greySquare.area();
24           System.out.println("Yellow Space: " + yellowSpace);
25       }
26   }
