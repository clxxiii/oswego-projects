1    /* 
2     * A class to calculate the maximum length of a rod  
3     * able to fit into a three-dimensional container 
4     */
5    package shapes;
6    
7    import java.util.Scanner;
8    
9    public class ShippingContainer {
10       public static void main(String[] args) {
11           // Get all the dimensions of the container
12           int[] containerDimensions = getContainerDimensions();
13           int containerHeight = containerDimensions[0];
14           int containerWidth = containerDimensions[1];
15           int containerLength = containerDimensions[2];
16   
17           // Calculate and construct the curtain
18           SRectangle containerEyeView = new SRectangle(containerLength, containerWidth);
19           double curtainLength = containerEyeView.diagonal();
20           SRectangle curtain = new SRectangle(containerHeight, curtainLength);
21   
22           //Calculate the length of the rod and print it
23           double rodLength = curtain.diagonal();
24           System.out.println("\nMaximum length of a rod inside the container: " + rodLength);
25       }
26   
27       /* 
28        * Fetch container dimensions from a scanner 
29        */
30       private static int[] getContainerDimensions() {
31           Scanner scanner = new Scanner(System.in);
32           System.out.print("Please enter three integers for the dimsions of the container (HxWxL): ");
33           int[] containerDimensions = new int[3];
34           containerDimensions[0] = scanner.nextInt();
35           containerDimensions[1] = scanner.nextInt();
36           containerDimensions[2] = scanner.nextInt();
37           return containerDimensions;
38       }
39   }
