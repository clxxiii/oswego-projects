4    package shapes;
5    
6    public class WorkSpace {
7    
8        public static void main(String[] args) {
9            double deskHeight = 66.0;
10           double deskWidth = 152.0;
11           double notebookHeight = 21.0;
12           double notebookWidth = 29.7;
13           double labManualHeight = 25.4;
14           double labManualWidth = 30.48;
15           double canRadius = 3.175;
16           double plateDiameter = 20;
17   
18           SRectangle desk = new SRectangle(deskHeight,deskWidth);
19           SRectangle notebook = new SRectangle(notebookHeight, notebookWidth);
20           SRectangle labManual = new SRectangle(labManualHeight, labManualWidth);
21           SCircle can = new SCircle(canRadius);
22           SSquare coaster = can.circumscribingSquare();
23           SCircle plates = new SCircle(plateDiameter / 2);
24   
25           double notebookSpace = notebook.area() * 2;
26           double labManualSpace = labManual.area();
27           double coasterSpace = coaster.area() * 3;
28           double platesSpace = plates.area() * 6;
29   
30           double coveredDeskSpace = notebookSpace + labManualSpace + coasterSpace + platesSpace;
31           double availableDeskSpace = desk.area() - coveredDeskSpace;
32           System.out.println("Available Desk Space = " + availableDeskSpace);
33       }
34   }
