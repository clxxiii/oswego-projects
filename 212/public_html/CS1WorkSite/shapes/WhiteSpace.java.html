2     * A program to computate the white space on a six sided die 
3     */
4    package shapes;
5    
6    public class WhiteSpace {
7        public static void main(String[] args) {
8            // Establish Givens
9            double dieEdgeLength = 0.63;
10   
11           // Calculate the surface area of the die
12           SSquare dieFace = new SSquare(dieEdgeLength);
13           double dieSurfaceArea = ( dieFace.area() * 6 );
14   
15           // Calculate the surface area of all the pips
16           double pipDiameter = dieEdgeLength / 6;
17           SCircle pip = new SCircle(pipDiameter / 2);
18           int numberOfPips = 1 + 2 + 3 + 4 + 5 + 6;
19           double pipSurfaceArea = pip.area() * numberOfPips;
20   
21           // Calculate and print the white space
22           double whiteSpace = dieSurfaceArea - pipSurfaceArea;
23           System.out.println("Area of the white space on the die: " + whiteSpace);
24       }
25   }
