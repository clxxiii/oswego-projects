15           private static void drawRectangle(int nrOfRows, int nrOfColumns) {
16               int i = 1;
17               while (i <= nrOfRows) {
18                   drawOneRow(nrOfColumns);
19                   i = i + 1;
20               }
21           }
22   
23       private static void drawOneRow(int nrOfColumns) {
24           int i = 1;
25           while (i <= nrOfColumns) {
26               System.out.print('*');
27               i++;
28           }
29           System.out.println("");
30       }
31   
32       private static int[] getRectangleDimensions() {

36           dimensions[0] = sc.nextInt();

38           dimensions[1] = sc.nextInt();
