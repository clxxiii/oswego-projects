6    package npw;
7    
8    import painter.SPainter;
9    import shapes.SCircle;
10   
11   import javax.swing.*;
12   import java.awt.*;
13   
14   public class BlueDot {
15       // Blue dot code
16       private void paintTheImage() {
17           SPainter klee = new SPainter("Blue Dot", 600, 600);
18           SCircle dot = new SCircle(200);
19           klee.setColor(Color.BLUE);
20           klee.paint(dot);
21       }
22   
23       // call paintTheImage() upon construction
24       public BlueDot() {
25           paintTheImage();
26       }
27   
28       // Construct a blue dot when calling main()
29       public static void main(String[] args) {
30           SwingUtilities.invokeLater(new Runnable() {
31               public void run() {
32                   new BlueDot();
33               }
34           });
35       }
36   }
