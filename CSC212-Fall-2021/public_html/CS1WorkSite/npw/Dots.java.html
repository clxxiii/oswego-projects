1    package npw;

3    import painter.SPainter;
4    import shapes.SCircle;
5    
6    import javax.swing.*;
7    import java.awt.*;
8    
9    public class Dots {
10       public void paintTheImage() {
11           SPainter painter = new SPainter("Dots", 600, 600);
12           SCircle dot = new SCircle(0);
13           paintRowOfCircles(painter, dot, Color.CYAN,25, -100, 75);
14           paintRowOfCircles(painter, dot, Color.YELLOW,30, 0, 100);
15           paintRowOfCircles(painter, dot, Color.MAGENTA,35, 100, 150);
16       }
17       public void paintRowOfCircles(SPainter painter, SCircle dot, Color color,int radius, double lat, double spread) {
18           dot.setRadius(radius);
19           painter.mfd(lat);
20           painter.setColor(color);
21           painter.paint(dot);
22           painter.setHeading(90.0);
23           painter.mfd(spread);
24           painter.paint(dot);
25           painter.mbk(spread*2);
26           painter.paint(dot);
27           painter.mfd(spread);
28           painter.faceNorth();
29           painter.mbk(lat);
30       }
31       public Dots() { paintTheImage(); }
32   
33       public static void main( String[] args) {
34           SwingUtilities.invokeLater(new Runnable() {
35               @Override
36               public void run() {
37                   new Dots();
38               }
39           });
40       }
