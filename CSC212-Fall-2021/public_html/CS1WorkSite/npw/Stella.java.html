1    package npw;
2    
3    import java.util.Scanner;
4    import java.awt.Color;
5    
6    import javax.swing.JOptionPane;
7    import javax.swing.SwingUtilities;
8    
9    import painter.SPainter;
10   import shapes.SSquare;
11   
12   public class Stella {
13       public static void main(String[] args) {
14           SwingUtilities.invokeLater(Stella::new);
15       }
16   
17       public Stella() {
18           paintSetup();
19       }
20   
21       private void paintSetup() {
22           int nrOfSquares = getNumber("how many squares");
23           Color outsideColor = new Color(getRandom(255), getRandom(255), getRandom(255));
24           Color insideColor = new Color(getRandom(255), getRandom(255), getRandom(255));
25           int canvasSize = 800;
26           int largestSquareSize = 700;
27   
28           SPainter painter = new SPainter("Stella Squares", canvasSize, canvasSize);
29           SSquare square = new SSquare(largestSquareSize);
30           
31           paintTheImage(painter, square, nrOfSquares, insideColor, outsideColor);
32           
33       }
34       
35       private void paintTheImage(SPainter painter, SSquare square, int nrOfSquares, Color insideColor,
36       Color outsideColor) {
37           
38           int currentSquareSize = (int) square.side();
39           double squareDecreasingSize = (double) square.side() / nrOfSquares;
40   
41           for(int i = 0; currentSquareSize > 0; i++) {
42               Color nextColor = nextColor(i, insideColor, outsideColor);
43               paintSquare(painter, square, nextColor);
44               currentSquareSize = currentSquareSize - (int) (squareDecreasingSize);
45               square.shrink(squareDecreasingSize);
46           }
47       }
48   
49       private void paintSquare(SPainter painter, SSquare square, Color nextColor) {
50           painter.setColor(nextColor);
51           painter.paint(square);
52       }
53   
54       private Color nextColor(int i, Color insideColor, Color outsideColor) {
55           if (i % 2 == 0) {
56               return outsideColor;
57           }
58           return insideColor;
59       }
60   
61       private int getRandom(int i) {
62           return (int) Math.floor(Math.random() * i);
63       }
64   
65       private int getNumber(String message) {
66           String nss = JOptionPane.showInputDialog(null, message + "?");
67           Scanner sc = new Scanner(nss);
68           int num = sc.nextInt();
69           sc.close();
70           return num;
71       }
72   }
