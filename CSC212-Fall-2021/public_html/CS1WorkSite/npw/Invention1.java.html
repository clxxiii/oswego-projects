25        * I have an idea for the design 
26        *  
27        * Basically, Make squares of side length 300, 280, 260, etc. All Inscribed. 
28        * By inscribed, i mean they are rotated so the corners touch the outside square's side. 
29        */
30       private void paintTheImage() {
31           
32           for (int i = 250; i > 0; i = i - 2) {
33               painter.setColor(getColor(i));
34               square.shrink(2);
35               painter.paint(square);
36           }
37   
38           for (int i = 0; i < 250; i = i + 20) {
39               painter.setColor(getColor(i));
40               painter.paint(circle);
41               circle.shrink(1);
42           }
43       }
44       private Color getColor(int i) {
45           return new Color(i, 255, 255);
46       }
