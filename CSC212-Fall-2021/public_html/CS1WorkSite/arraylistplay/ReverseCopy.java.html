1    /* 
2     * Takes a file, reverses the order of the words in the file, and generates a new file. 
3     */
4    package arraylistplay;
5    
6    import java.io.File;
7    import java.io.FileNotFoundException;
8    import java.io.IOException;
9    import java.io.PrintWriter;
10   import java.util.ArrayList;
11   import java.util.Scanner;
12   
13   public class ReverseCopy {
14       public static void main(String[] args) throws FileNotFoundException, IOException {
15           String inputFileName = "WordsINeverSaid.text";
16           String outputFileName = "WordsINeverSaidReversed.text";
17           createFullFileName(inputFileName);
18           ArrayList<String> words = readWordsFromFileName(inputFileName);
19           writeWordsToFile(words,outputFileName);
20       }
21   
22       private static ArrayList<String> readWordsFromFileName(String inputFileName) throws FileNotFoundException {
23           // Equate a scanner with the input file
24           Scanner scanner = establishScanner(inputFileName);
25           ArrayList<String> words = new ArrayList<>();
26           while (scanner.hasNext()) {
27               String word = scanner.next();
28               words.add(word);
29           }
30           // Return the words
31           return words;
32       }
33   
34       private static void writeWordsToFile(ArrayList<String> words, String outputFileName) throws IOException {
35           // Equate a printer with an output file
36           PrintWriter printer = getPrintWriter(outputFileName);
37           // Print the words to the file
38           for ( int x = words.size() - 1; x >= 0; x--) {
39               printer.println(words.get(x));
40           }
41           printer.close();
42       }
43   
44       private static PrintWriter getPrintWriter(String outputFileName) throws FileNotFoundException {
45           String fullFileName = createFullFileName(outputFileName);
46           PrintWriter printer = new PrintWriter(fullFileName);
47           return printer;
48       }
49   
50       private static Scanner establishScanner(String inputFileName) throws FileNotFoundException {
51           String fullFileName = createFullFileName(inputFileName);
52           return new Scanner(new File(fullFileName));
53       }
54       // Create the full file name for a simple file name, assuming that it will be
55       // found in the CS1Files/data subdirectory of the user's home directory.
56       private static String createFullFileName(String fileName) {
57           String separator = System.getProperty("file.separator");
58           String home = System.getProperty("user.home");
59           String path = home + separator + "CS1Files" + separator + "data" + separator;
60           String fullFileName = path + fileName;
61           return fullFileName;
62       }
63   }
